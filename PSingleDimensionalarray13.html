13. Enter an array from user and count the duplicate values?

Code:

        #include 


            int main() {
                int arr[50];
                int n, i, j, count;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                int counted[50] = {0}; 
                for (i = 0; i < n; i++) 
                {
                    if (counted[i] == 0) { 
                        count = 1;
                        for (j = i + 1; j < n; j++) {
                            if (arr[i] == arr[j]) {
                                count++; 
                                counted[j] = 1; 
                            }
                        }
                            if (count > 1) { 
                            printf("Value %d occurs %d times\n", arr[i], count);
                        }
                    }
                }
            
                return 0;
            }
