19. Find the local maxima in the array?

Code:

        #include 


            int main() {
                int arr[100];
                int n;
            
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
                
                printf("Enter %d elements:\n", n);
                for (int i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
                printf("Local maxima in the array are:\n");
                for (int i = 0; i < n; i++) {
                  
                    if ((i == 0 && n > 1 && arr[i] > arr[i + 1]) || 
                        (i == n - 1 && n > 1 && arr[i] > arr[i - 1]) || 
                        (i > 0 && i < n - 1 && arr[i] > arr[i - 1] && arr[i] > arr[i + 1])) { 
                        printf("%d ", arr[i]);
                    }
                }
            
                printf("\n");
                return 0; 
            }
