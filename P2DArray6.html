6. Enter the matrix from user, multiply it with a scalar value and display the answer.

Code:

        #include 

            int main() {
                int rows, cols, scalar;
                            
                printf("Enter the number of rows: ");
                scanf("%d", &rows);
                printf("Enter the number of columns: ");
                scanf("%d", &cols);
                            
                int matrix[rows][cols];
                            
                printf("Enter elements of the matrix:\n");
                for (int i = 0; i < rows; i++) {
                    for (int j = 0; j < cols; j++) {
                        printf("Enter element at position [%d][%d]: ", i, j);
                        scanf("%d", &matrix[i][j]);
                    }
                }
                            
                printf("Enter the scalar value to multiply: ");
                scanf("%d", &scalar);
                            
                for (int i = 0; i < rows; i++) {
                    for (int j = 0; j < cols; j++) {
                    matrix[i][j] *= scalar;
                    }
                }
                printf("Resultant matrix after multiplication by scalar %d:\n", scalar);
                for (int i = 0; i < rows; i++) {
                    for (int j = 0; j < cols; j++) {
                        printf("%d ", matrix[i][j]);
                    }
                    printf("\n");
                }
                            
                return 0;
            }
