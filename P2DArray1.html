1. Enter a matrix of 4 rows and 2 columns from the user display it in matrix format.

Code:

        #include < stdio.h >

        int main() {
        int matrix[4][2];
        printf("Enter elements for a 4x2 matrix:\n");
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 2; j++) {
                printf("Enter element at [%d][%d]: ", i, j);
                scanf("%d", &matrix[i][j]);
            }
        }
        printf("\nThe 4x2 matrix is:\n");
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 2; j++) {
                printf("%d ", matrix[i][j]);
            }
            printf("\n");
        }
        return 0;
        }
