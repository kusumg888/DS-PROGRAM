5. Enter an integer array from the user, enter an integer value and perform binary search algorithm through iteration.

Code:

        #include 

            void search(int arr[], int size, int value) {
                int i;
                
                printf("\n");
                for (i = 0; i < size; i++) {
                    if (arr[i] == value) {
                        printf("Value %d is present at index %d\n", value, i);
                        return; 
                    }
                }
                printf("-1 (Value not found)\n");
            }
            
            int main() {
                int y, s, i;
                printf("Enter the size of the array: ");
                scanf("%d", &s);
                int arr[s];
                printf("Enter the values in the array: \n");
                for (i = 0; i < s; i++) {
                    scanf("%d", &arr[i]);
                }
                printf("Enter the value to search: ");
                scanf("%d", &y);
                search(arr, s, y);
                return 0; 
            }
