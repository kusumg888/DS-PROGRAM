7. Implement the operations of priority queue in an integer array (insert, delete, peek, isEmpty, isFull).

Code:

        #include 
            #include 
            
            #define MAX 5
            
            struct PriorityQueue {
                int items[MAX];
                int size;
            };
            
            struct PriorityQueue* createQueue() {
                struct PriorityQueue* pq = (struct PriorityQueue*)malloc(sizeof(struct PriorityQueue));
                pq->size = 0;
                return pq;
            }
            
            int isEmpty(struct PriorityQueue* pq) {
                return pq->size == 0;
            }
            
            int isFull(struct PriorityQueue* pq) {
                return pq->size == MAX;
            }
            
            void insert(struct PriorityQueue* pq, int value) {
                if (isFull(pq)) {
                    printf("Queue is full. Cannot insert %d.\n", value);
                    return;
                }
                int i;
                for (i = pq->size - 1; (i >= 0 && pq->items[i] < value); i--) {
                    pq->items[i + 1] = pq->items[i];
                }
                pq->items[i + 1] = value;
                pq->size++;
                printf("Inserted %d\n", value);
            }
            
            int delete(struct PriorityQueue* pq) {
                if (isEmpty(pq)) {
                    printf("Queue is empty. Cannot delete.\n");
                    return -1;
                }
                return pq->items[--pq->size];
            }
            
            int peek(struct PriorityQueue* pq) {
                if (isEmpty(pq)) {
                    printf("Queue is empty. Cannot peek.\n");
                    return -1;
                }
                return pq->items[pq->size - 1];
            }
            
            int main() {
                struct PriorityQueue* pq = createQueue();
                int choice, value;
            
                while (1) {
                    printf("\n1. Insert\n2. Delete\n3. Peek\n4. Exit\n");
                    printf("Enter your choice: ");
                    scanf("%d", &choice);
            
                    switch (choice) {
                        case 1:
                            printf("Enter value to insert: ");
                            scanf("%d", &value);
                            insert(pq, value);
                            break;
                        case 2:
                            value = delete(pq);
                            if (value != -1) printf("Deleted: %d\n", value);
                            break;
                        case 3:
                            value = peek(pq);
                            if (value != -1) printf("Highest priority element: %d\n", value);
                            break;
                        case 4:
                            free(pq);
                            exit(0);
                        default:
                            printf("Invalid choice.\n");
                    }
                }
                return 0;
            }
