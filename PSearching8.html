            void bubbleSort(char str[]) {
                int n = strlen(str);
                for (int i = 0; i < n - 1; i++) {
                    for (int j = 0; j < n - i - 1; j++) {
                        if (str[j] > str[j + 1]) {
                       
                            char temp = str[j];
                            str[j] = str[j + 1];
                            str[j + 1] = temp;
                        }
                    }
                }
            }
            
            int binarySearch(char str[], char target) {
                int left = 0;
                int right = strlen(str) - 1;
            
                while (left <= right) {
                    int mid = left + (right - left) / 2;
            
                    if (str[mid] == target) {
                        return mid; 
                    }
                   
                    if (str[mid] > target) {
                        right = mid - 1;
                    } else { 
                        left = mid + 1;
                    }
                }
                return -1; 
            }
            
            int main() {
                char str[100]; 
                char target;
            
                printf("Enter a string: ");
                fgets(str, sizeof(str), stdin); 
            
                str[strcspn(str, "\n")] = 0;
            
                bubbleSort(str);
                printf("Sorted string: %s\n", str);
            
                printf("Enter a character to search for: ");
                scanf(" %c", &target); 
                int result = binarySearch(str, target);
            
                if (result != -1) {
                    printf("Character '%c' found at index %d.\n", target, result);
                } else {
                    printf("Character '%c' not found in the string.\n", target);
                }
            
                return 0;
