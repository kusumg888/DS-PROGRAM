4. Enter an array from user and copy it into another array in reverse order?

Code:

        #include 

            int main() {
                int arr1[50]; 
                int arr2[50]; 
                int n, i,j;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr1[i]);
                }
                j=n-1;
                for (i = 0; i < n; i++) {
                    
                    arr2[j] = arr1[i];
                    j--;
                }
            
                printf("The elements of the second array are:\n");
                for (j = 0; j < n; j++) {
                    printf("%d ", arr2[j]);
                }
                printf("\n"); 
                return 0; 
            }
