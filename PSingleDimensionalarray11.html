11. Enter an array from user and find the duplicate values?

Code:

        #include 

            int main() {
                int arr[50]; 
                int n, i,j;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
                
                printf("The dublicate value of the array are:\n");
                for (i= 0; i < n; i++) {
                    for (j= i+1; j < n; j++)
                    {
                       if(arr[i]==arr[j])
                      {
                        printf("%d ",arr[i]);
                        break;
                      }
                    }
                }
                 
                printf("\n"); 
                return 0; 
            }
