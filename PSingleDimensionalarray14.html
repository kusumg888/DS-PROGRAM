14. Enter an array from user and print all possible combinations of the elements?

Code:

        #include 

            void printCombinations(int arr[], int n, int index, int combination[], int combIndex) {
              
                if (combIndex > 0) { 
                    for (int i = 0; i < combIndex; i++) {
                        printf("%d ", combination[i]);
                    }
                    printf("\n");
                }
            
                for (int i = index; i < n; i++) {
                    combination[combIndex] = arr[i]; 
                    printCombinations(arr, n, i + 1, combination, combIndex + 1); 
                }
            }
            
            int main() {
                int arr[50];
                int n;
            
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
            
                printf("Enter %d elements:\n", n);
                for (int i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                int combination[50]; 
                printf("All possible combinations are:\n");
                printCombinations(arr, n, 0, combination, 0);
                return 0; 
            }
