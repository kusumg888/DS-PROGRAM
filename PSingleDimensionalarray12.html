12. Enter an array from user and remove the duplicate values and fill with zero?

Code:

        #include 

            int main() {
                int arr[50]; 
                int n, i,j;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
                
              
                for (i= 0; i < n; i++) {
                    for (j= i+1; j < n; j++)
                    {
                       if(arr[i]==arr[j])
                      {
                          arr[j]=0;
                      }
                    }
                }
                 printf("The array after removing duplicates and filling with zeros is:\n");
                for (i = 0; i < n; i++) {
                    printf("%d ", arr[i]);
                }
                
                 
                printf("\n"); 
                return 0; 
            }
