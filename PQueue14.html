            int longestValidParentheses(char *s) {
                int maxLength = 0;
                int *stack = (int *)malloc(sizeof(int) * (strlen(s) + 1));
                int top = -1;
            
                stack[++top] = -1;
            
                for (int i = 0; s[i] != '\0'; i++) {
                    if (s[i] == '(') {
                        stack[++top] = i;
                    } else {
                        top--;
                        if (top == -1) {
                         
                            stack[++top] = i;
                        } else {
                            int length = i - stack[top];
                            if (length > maxLength) {
                                maxLength = length;
                            }
                        }
                    }
                }
            
                free(stack);
                return maxLength;
            }
