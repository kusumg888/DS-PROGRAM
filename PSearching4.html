4. Enter a string from the user, enter a character and perform linear search algorithm through iteration.

Code:

        #include 

            void linearSearch(char str[], char target) {
                int found = 0; 
                int index = -1; 
                for (int i = 0; str[i] != '\0'; i++) {
                    if (str[i] == target) {
                        found = 1; 
                        index = i; 
                        break; 
                    }
                }
            
                // Output the result
                if (found) {
                    printf("Character '%c' found at index %d.\n", target, index);
                } else {
                    printf("Character '%c' not found in the string.\n", target);
                }
            }
            
            int main() {
                char str[100]; 
                char target;
            
                printf("Enter a string: ");
                gets(str); 
                printf("Enter a character to search for: ");
                scanf(" %c", &target); 
                linearSearch(str, target);
                
                return 0; 
            }
