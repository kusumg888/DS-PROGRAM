2. Enter an array from user and print the array in reverse order?

Code:

        #include 


            int main() {
                int arr[50];
                int n, i;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                printf("The elements of the array in reverse order are:\n");
                for (i = n - 1; i >= 0; i--) {
                    printf("%d ", arr[i]);
                }
                printf("\n"); 
                return 0; 
            }
