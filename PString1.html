1. Enter the string from user and convert in uppercase.

Code:

        #include 

            int main() {
                char string[100];
            
                printf("Enter a string: ");
                gets(string);
              
                for (int i = 0; string[i] != '\0'; i++) 
                {
                    if (string[i] >= 'a' && string[i] <= 'z') {
                     
                       string[i] = string[i] - 32;
                    }
                }
                printf("Uppercase string: %s", string); 
            
                return 0;
            }
