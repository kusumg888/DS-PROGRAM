3. Enter an array from user and copy the array into second array?

Code:

        #include 

            int main() {
                int arr1[50]; 
                int arr2[50]; 
                int n, i;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr1[i]);
                }
                for (i = 0; i < n; i++) {
                    arr2[i] = arr1[i];
                }
            
                printf("The elements of the second array are:\n");
                for (i = 0; i < n; i++) {
                    printf("%d ", arr2[i]);
                }
                printf("\n"); 
                return 0; 
            }
