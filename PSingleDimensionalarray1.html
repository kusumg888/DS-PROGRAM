                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                printf("The elements of the array are:\n");
                for (i = 0; i < n; i++) {
                    printf("%d ", arr[i]);
                }
                printf("\n"); 
                return 0;
            }
