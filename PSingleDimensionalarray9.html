9. Enter an array from user and find the k-th smallest value?

Code:

        #include 

            void bubbleSort(int arr[], int n) {
                for (int i = 0; i < n - 1; i++) {
                    for (int j = 0; j < n - i - 1; j++) 
                    {
                        if (arr[j] > arr[j + 1]) 
                        {
                            int temp = arr[j];
                            arr[j] = arr[j + 1];
                            arr[j + 1] = temp;
                        }
                    }
                }
            }
            
            int main() {
                int arr[50];
                int n, k;
            
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
            
                printf("Enter %d elements:\n", n);
                for (int i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                printf("Enter the value of k (1 to %d): ", n);
                scanf("%d", &k);
                bubbleSort(arr, n);
                printf("The %d-th smallest value is: %d\n", k, arr[k - 1]);
            
                return 0;
            }
