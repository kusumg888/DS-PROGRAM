10. Enter an array from user and add all values?

            int main() {
                int arr[50]; 
                int n, i,sum=0;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
                
                printf("The addition of all the array are:\n");
                for (i= 0; i < n; i++) {
                    sum+=arr[i];
                }
                 printf("%d ", sum);
                printf("\n"); 
                return 0; 
            }
