2. Enter a string from the user, enter a character and perform linear search algorithm through recursion.

Code:

        #include 

            int linearSearch(char str[], char target, int index) {
                
                if (str[index] == '\0') {
                    return -1; 
                }
                if (str[index] == target) {
                    return index; 
                }
                return linearSearch(str, target, index + 1);
            }
            
            int main() {
                char str[100]; 
                char target;
                printf("Enter a string: ");
                fgets(str, sizeof(str), stdin); 
                printf("Enter a character to search for: ");
                scanf(" %c", &target); 
                int result = linearSearch(str, target, 0);
            
                if (result != -1) {
                    printf("Character '%c' found at index %d.\n", target, result);
                } else {
                    printf("Character '%c' not found in the string.\n", target);
                }
                return 0; 
            }
