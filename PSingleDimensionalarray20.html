20. Trapping rain water problem: Given n non-negative integers representing an elevation map where the width of each bar is 1, compute how much water it can trap after raining?

Code:

        #include 

            int trap(int height[], int n) {
                if (n <= 2) return 0; 
            
                int left = 0, right = n - 1;
                int leftMax = height[left];
                int rightMax = height[right];
                int waterTrapped = 0;
            
                while (left < right) {
                    if (height[left] < height[right]) {
                        left++; 
                        if (height[left] < leftMax) {
                            waterTrapped += leftMax - height[left];
                        } else {
                            leftMax = height[left]; 
                        }
                    } else {
                        right--
                        if (height[right] < rightMax) {
                            waterTrapped += rightMax - height[right]; 
                        } else {
                            rightMax = height[right]; 
                        }
                    }
                }
            
                return waterTrapped;
            }
            
            int main() {
                int n;
                printf("Enter the number of elements in the elevation map: ");
                scanf("%d", &n);
                int height[n];
                printf("Enter the heights:\n");
                for (int i = 0; i < n; i++) {
                    scanf("%d", &height[i]);
                }
            
                int result = trap(height, n);
                printf("The amount of water that can be trapped is: %d\n", result);
            
                return 0; 
            }
